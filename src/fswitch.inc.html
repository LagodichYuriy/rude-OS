  5 _switch:
  6         call _screen_clean      ; clean the screen
  7         call _cursor_reset      ; reset cursor position
  8 
  9         xor ax, ax
 10         mov ds, ax              ; update data segment
 11 
 12         cli                     ; disable interrupts
 13         lgdt [gdtr]             ; load GDT register with start address of GDT
 14 
 15         or   al,2               ; enable A20
 16         out  0x60, al
 17 
 18         mov  al,0xAE            ; enable keyboard
 19         out  0x64,al
 20 
 21         mov  eax,cr0
 22         or   al,1               ; set PE (Protection Enable) bit in CR0
 23         mov  cr0,eax
 24 
 25         jmp (CODE_DESC - NULL_DESC) : _protected_mode

 29 ; ====================================== ;
 30 ; ▶ Global Descriptor Table (GDT)        ;
 31 ; ====================================== ;
 32 
 33 NULL_DESC:
 34     dd 0                        ; null descriptor
 35     dd 0
 36 
 37 CODE_DESC:
 38     dw 0xFFFF                   ; limit low
 39     dw 0                        ; base low
 40     db 0                        ; base middle
 41     db 10011010b                ; access
 42     db 11001111b                ; granularity
 43     db 0                        ; base high
 44 
 45 DATA_DESC:
 46     dw 0xFFFF                   ; data descriptor
 47     dw 0                        ; limit low
 48     db 0                        ; base low
 49     db 10010010b                ; access
 50     db 11001111b                ; granularity
 51     db 0                        ; base high
 52 
 53 gdtr:
 54     limit dw 24                 ; length of GDT
 55     base  dd NULL_DESC          ; base of GDT
 56 
 57 
 58 
 59 ; ====================================== ;
 60 ; ▶ Protected mode segment               ;
 61 ; ====================================== ;
 62 
 63 use32
 64 
 65 _protected_mode:
 66         mov  edi,0xb8000        ; direct VGA address (text-mode)
 67 
 68         mov  eax,msg_prot       ; save message adress to `eax`
 69         call __print_message    ; print a message
 70 
 71         call __keyboard_loop    ; wait for a `Enter` key be pressed
 72 
 73 
 74 ; ====================================== ;
 75 ; ▶ Real fallback mode segment           ;
 76 ; ====================================== ; 
 77 
 78 __real_mode:
 79         cli                     ; Disable interrupts.
 80 
 81         ; need 16-bit Protected Mode GDT entries
 82         mov eax,gdtr            ; 16-bit Protected Mode data selector.
 83         mov ds,ax
 84         mov es,ax
 85         mov fs,ax
 86         mov gs,ax
 87 
 88         ; disable paging (we need everything to be 1:1 mapped).
 89         mov eax, cr0            ; copy `cr0` in `eax`
 90         mov [savcr0], eax       ; save pmode CR0
 91         and eax, 0x7FFFFFFe     ; disable paging bit & enable 16-bit pmode
 92         mov cr0, eax
 93 
 94         jmp 0:__real_mode       ; perform Far jump to set CS
 95 
 96         mov sp, 0x8000          ; pick a stack pointer
 97         mov ax, 0               ; reset segment registers to 0
 98         mov ds, ax              ;
 99         mov es, ax              ;
100         mov fs, ax              ;
101         mov gs, ax              ;
102         lidt [idtr]             ; define a new interrupt table
103 
104         sti                     ; restore interrupts
105 
106 __real_mode_loop:
107         ; loop is here
108         jmp __real_mode_loop
109 
110 
111 
112 ; ====================================== ;
113 ;       ↓ Keyboard loop procedure ↓      ;
114 
115 __keyboard_loop:
116 ;       push bp ax bx cx dx     ; protect registers against changes
117 
118         in   al,60h             ; read keyboard scan code
119 
120 
121         mov  dl,al              ; make a copy of scancode in `dl` register
122 
123 
124         xor  bh,bh              ; clean the `bh` register
125         mov  bl,al              ; update keyboard state
126         and  bl,7Fh             ; bx = scan code
127         shr  al,7               ; al = 0 if pressed, 1 if released
128         xor  al,1               ; al = 1 if pressed, 0 if released
129 
130 
131         mov  cl,al              ; make a copy of keypress event in `cl`
132 
133 
134         in   al,61h             ;
135         mov  ah,al              ;
136         or   al,80h             ; send EOI to XT keyboard
137         out  61h,al             ;
138         mov  al,ah              ;
139         out  61h,al             ;
140 
141         mov  al,20h             ; place `EOI` in the `al` register
142         out  20h,al             ; send `EOI` to master PIC
143 
144         cmp  cl,1               ; is key pressed?
145         jne  __keyboard_loop    ; wait for another keypress if not
146 
147         cmp  dl,1Ch             ; is it `Enter` key?
148         jne  __keyboard_loop    ; wait for another keypress if not
149 
150 ;       pop  dx cx bx ax bp     ; restore origin values from the stack
151 
152         ret
153 
154 ;       ↑ Keyboard loop procedure ↑      ;
155 ; ====================================== ;
156 ; -------------------------------------- ;
157 ; ====================================== ;
158 ;      ↓ Print a message procedure ↓     ;
159 
160 __print_message:
161         mov  dl,byte[eax]
162 
163         cmp  dl,0
164         je   __print_message_end
165         call __print_letter
166 
167         inc  ax
168         jmp __print_message
169 
170 __print_message_end:
171         ret
172 
173 ;      ↑ Print a message procedure ↑     ;

177 ;      ↓ Print a letter procedure ↓      ;
178 
179 __print_letter:
180         mov byte[edi],dl
181         inc edi
182 
183         mov byte[edi],0x07
184         inc edi
185 
186         ret

199 idtr:
200         dw 0x3ff                ; 256 entries, 4b each = 1K
201         dd 0                    ; Real Mode IVT @ 0x0000
202 
203 savcr0:
204         dd 0                    ; Storage location for pmode cr0.
